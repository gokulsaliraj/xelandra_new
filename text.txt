<script>
  gsap.registerPlugin(ScrollTrigger);

  // Heading
  gsap.from(".about-text h2", {
    x: -80,
    opacity: 0,
    duration: 1,
    ease: "power3.out",
    scrollTrigger: {
      trigger: ".about-text h2",
      start: "top 80%",   // when heading is 80% into viewport
      toggleActions: "play reverse play reverse"
    }
  });

  // Paragraph
  gsap.from(".about-text p", {
    x: -80,
    opacity: 0,
    duration: 1,
    ease: "power3.out",
    scrollTrigger: {
      trigger: ".about-text p",
      start: "top 85%",
      toggleActions: "play reverse play reverse"
    }
  });

  // Features
  gsap.from(".about-features .feature", {
    x: -50,
    opacity: 0,
    duration: 0.8,
    stagger: 0.2,
    ease: "back.out(1.7)",
    scrollTrigger: {
      trigger: ".about-features",
      start: "top 85%",
      toggleActions: "play reverse play reverse"
    }
  });

  // Image

    x: 100,

    ease: "power3.out",

  });
